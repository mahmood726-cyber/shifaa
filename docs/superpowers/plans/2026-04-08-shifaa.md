# Shifaa Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the Shifaa Global Healing Equity Intelligence engine — linking IHME GBD burden, CT.gov trials, World Bank socioeconomics, and WHO health systems to compute a Research Equity Index for every country-disease pair, producing 5 E156 papers and an interactive HTML dashboard.

**Architecture:** Pure analysis project that reads 3 existing lakehouses (Parquet) and CT.gov API. Builds a single `shifaa_matrix` table (country × disease × year) that all 5 papers query. Multilevel Poisson regression as the core causal model. Single-file HTML dashboard with 5 panels.

**Tech Stack:** Python 3.11+, pandas, pyarrow, requests, statsmodels, scipy, pytest

**Spec:** `docs/superpowers/specs/2026-04-08-shifaa-design.md`

**Data source schemas (verified):**
- WB harmonized: `iso3c, year, indicator_code, indicator_name, value, lower, upper, sex, age_group`
- IHME native (dalys.parquet): `measure_id, location_id, location_name, sex_id, sex_name, age_id, age_name, cause_id, cause_name, year, val, upper, lower, metric_id, metric_name`
- WHO World Health Stats: `ind_code, dim_geo_code, dim_time_year, value_numeric`
- WHO HIDR: `iso3, indicator_abbr, estimate, ...`
- CT.gov API v2: nested JSON with `protocolSection.conditionsModule.conditions[]` and `protocolSection.contactsLocationsModule.locations[].country`

```python
# src/shifaa/crosswalk/loader.py
"""Load and query the GBD↔MeSH↔ICD crosswalk."""
from __future__ import annotations

from pathlib import Path

import pandas as pd

from shifaa.config import CROSSWALK_PATH


def load_crosswalk(path: Path | None = None) -> pd.DataFrame:
    """Load the GBD-MeSH-ICD crosswalk CSV."""
    if path is None:
        path = CROSSWALK_PATH
    return pd.read_csv(path, dtype={"gbd_cause_id": int})


def match_condition(condition_text: str, crosswalk: pd.DataFrame) -> dict | None:
    """Match a CT.gov condition string to a GBD cause via MeSH terms.

    Uses case-insensitive substring matching against semicolon-delimited
    MeSH terms. Returns the first match as a dict, or None.
    """
    cond_lower = condition_text.lower()
    for _, row in crosswalk.iterrows():
        mesh_terms = str(row["mesh_terms"]).split(";")
        for term in mesh_terms:
            term_lower = term.strip().lower()
            if not term_lower:
                continue
            # Match if the MeSH term appears in the condition or vice versa
            if term_lower in cond_lower or cond_lower in term_lower:
                return {
                    "gbd_cause_id": int(row["gbd_cause_id"]),
                    "gbd_cause_name": row["gbd_cause_name"],
                    "matched_term": term.strip(),
                }
    return None


def match_conditions_batch(conditions: list[str], crosswalk: pd.DataFrame) -> list[dict | None]:
    """Match a list of CT.gov conditions. Returns list parallel to input."""
    return [match_condition(c, crosswalk) for c in conditions]
```

```python
# src/shifaa/ctgov/api.py
"""CT.gov API v2 client: search studies by condition, parse responses."""
from __future__ import annotations

import requests
from requests.adapters import HTTPAdapter
from urllib3.util.retry import Retry

from shifaa.config import CTGOV_BASE_URL, CTGOV_PAGE_SIZE, CTGOV_TIMEOUT

FIELDS = ",".join([
    "protocolSection.identificationModule.nctId",
    "protocolSection.identificationModule.briefTitle",
    "protocolSection.statusModule.overallStatus",
    "protocolSection.statusModule.startDateStruct",
    "protocolSection.conditionsModule.conditions",
    "protocolSection.designModule.enrollmentInfo",
    "protocolSection.contactsLocationsModule.locations",
])


def build_session() -> requests.Session:
    retry = Retry(total=3, backoff_factor=1, status_forcelist=(429, 500, 502, 503, 504))
    adapter = HTTPAdapter(max_retries=retry)
    session = requests.Session()
    session.headers.update({"User-Agent": "shifaa/0.1"})
    session.mount("https://", adapter)
    return session


def search_studies_by_condition(
    condition: str,
    session: requests.Session | None = None,
    max_pages: int = 50,
) -> list[dict]:
    """Search CT.gov for studies matching a condition. Returns parsed studies."""
    if session is None:
        session = build_session()

    params = {
        "query.cond": condition,
        "pageSize": CTGOV_PAGE_SIZE,
        "fields": FIELDS,
    }

    all_studies: list[dict] = []
    page_token = None

    for _ in range(max_pages):
        if page_token:
            params["pageToken"] = page_token

        resp = session.get(CTGOV_BASE_URL, params=params, timeout=CTGOV_TIMEOUT)
        resp.raise_for_status()
        data = resp.json()

        studies = data.get("studies", [])
        for study in studies:
            parsed = parse_study(study)
            if parsed:
                all_studies.append(parsed)

        page_token = data.get("nextPageToken")
        if not page_token:
            break

    return all_studies


def parse_study(study: dict) -> dict | None:
    """Parse a CT.gov API v2 study response into a flat dict."""
    proto = study.get("protocolSection", {})
    ident = proto.get("identificationModule", {})
    status = proto.get("statusModule", {})
    conds = proto.get("conditionsModule", {})
    design = proto.get("designModule", {})
    contacts = proto.get("contactsLocationsModule", {})

    nct_id = ident.get("nctId")
    if not nct_id:
        return None

    # Extract start year
    start_date = status.get("startDateStruct", {}).get("date", "")
    start_year = None
    if start_date and len(start_date) >= 4:
        try:
            start_year = int(start_date[:4])
        except ValueError:
            pass

    # Extract enrollment
    enrollment_info = design.get("enrollmentInfo", {})
    enrollment = enrollment_info.get("count")

    # Extract locations
    locations = []
    for loc in contacts.get("locations", []):
        if loc.get("country"):
            locations.append({
                "facility": loc.get("facility", ""),
                "city": loc.get("city", ""),
                "country": loc["country"],
            })

    return {
        "nct_id": nct_id,
        "title": ident.get("briefTitle", ""),
        "status": status.get("overallStatus", ""),
        "start_year": start_year,
        "enrollment": enrollment,
        "conditions": conds.get("conditions", []),
        "locations": locations,
    }
```

```python
    """Return the country name → iso3c mapping."""

def allocate_trial(trial: dict) -> dict[str, float]:
    """Compute site-weighted country allocation for a single trial.

    Returns {iso3c: weight} where weights sum to 1.0.
    Countries not in the mapping are skipped.
    """
    locations = trial.get("locations", [])
    if not locations:
        return {}

    # Count sites per country, map to iso3c
    country_counts: Counter[str] = Counter()
    for loc in locations:
        country_name = loc.get("country", "")
        iso3c = _COUNTRY_ISO3.get(country_name)
        if iso3c:
            country_counts[iso3c] += 1

    if not country_counts:
        return {}

    total = sum(country_counts.values())
    return {iso3c: count / total for iso3c, count in country_counts.items()}
```

```python
# src/shifaa/ctgov/fetch.py
"""Batch-fetch CT.gov trials for all crosswalk causes and build trial matrix."""
from __future__ import annotations

from pathlib import Path

import pandas as pd

from shifaa.config import CTGOV_CACHE_DIR
from shifaa.ctgov.allocate import allocate_trial
from shifaa.ctgov.api import build_session, search_studies_by_condition
from shifaa.crosswalk.loader import load_crosswalk, match_condition


def fetch_trials_for_cause(
    cause_row: dict,
    cache_dir: Path | None = None,
    session=None,
    max_pages: int = 50,
) -> dict:
    """Fetch all CT.gov trials for a single GBD cause.

    Uses the first MeSH term as the primary search query.
    Caches results as JSON.
    """
    if cache_dir is None:
        cache_dir = CTGOV_CACHE_DIR
    cache_dir.mkdir(parents=True, exist_ok=True)

    gbd_id = cause_row["gbd_cause_id"]
    mesh_terms = str(cause_row.get("mesh_terms", "")).split(";")
    primary_term = mesh_terms[0].strip() if mesh_terms else cause_row["gbd_cause_name"]

    cache_file = cache_dir / f"cause_{gbd_id}.json"

    # Use cache if exists
    if cache_file.exists():
        import json
        cached = json.loads(cache_file.read_text("utf-8"))
        return cached

    # Fetch from API
    if session is None:
        session = build_session()

    studies = search_studies_by_condition(primary_term, session=session, max_pages=max_pages)

    # Allocate each study to countries
    allocated = []
    for study in studies:
        alloc = allocate_trial(study)
        if alloc:
            allocated.append({
                "nct_id": study["nct_id"],
                "start_year": study["start_year"],
                "gbd_cause_id": gbd_id,
                "allocation": alloc,
            })

    result = {
        "gbd_cause_id": gbd_id,
        "gbd_cause_name": cause_row.get("gbd_cause_name", ""),
        "search_term": primary_term,
        "total_studies": len(studies),
        "allocated_studies": len(allocated),
        "trials": allocated,
    }

    # Cache
    import json
    cache_file.write_text(json.dumps(result, indent=2), encoding="utf-8")

    return result


def fetch_all_causes(
    crosswalk_path: Path | None = None,
    cache_dir: Path | None = None,
) -> list[dict]:
    """Fetch trials for all causes in the crosswalk. Returns list of cause results."""
    cw = load_crosswalk(crosswalk_path)
    session = build_session()
    results = []
    try:
        for _, row in cw.iterrows():
            result = fetch_trials_for_cause(row.to_dict(), cache_dir=cache_dir, session=session)
            results.append(result)
    finally:
        session.close()
    return results


def build_trial_matrix(allocated_trials: list[dict]) -> pd.DataFrame:
    """Build a trial density matrix from allocated trials.

    Returns DataFrame with columns: iso3c, year, gbd_cause_id, trial_density, trial_count_binary
    """
    rows: list[dict] = []
    for trial in allocated_trials:
        year = trial.get("start_year")
        gbd_id = trial.get("gbd_cause_id")
        alloc = trial.get("allocation", {})
        if not year or not gbd_id or not alloc:
            continue
        for iso3c, weight in alloc.items():
            rows.append({
                "iso3c": iso3c,
                "year": year,
                "gbd_cause_id": gbd_id,
                "trial_weight": weight,
            })

    if not rows:
        return pd.DataFrame(columns=["iso3c", "year", "gbd_cause_id", "trial_density", "trial_count_binary"])

    df = pd.DataFrame(rows)
    grouped = df.groupby(["iso3c", "year", "gbd_cause_id"]).agg(
        trial_density=("trial_weight", "sum"),
        trial_count_binary=("trial_weight", "count"),
    ).reset_index()

    return grouped
```

```python
# src/shifaa/lakehouse/reader.py
"""Read harmonized/native Parquet from IHME, WB, and WHO lakehouses."""
from __future__ import annotations

from pathlib import Path

import pandas as pd

from shifaa.config import IHME_SILVER, IHME_REFERENCE, WB_SILVER, WHO_SILVER, WB_INDICATORS


def read_wb_indicators(silver_dir: Path | None = None) -> pd.DataFrame | None:
    """Read all WB harmonized Parquet into a single long-format DataFrame.

    Returns DataFrame with columns: iso3c, year, indicator_code, value
    """
    if silver_dir is None:
        silver_dir = WB_SILVER

    frames = []
    for domain_dir in silver_dir.iterdir():
        harm_dir = domain_dir / "harmonized"
        if not harm_dir.is_dir():
            continue
        for pq in harm_dir.glob("*.parquet"):
            df = pd.read_parquet(pq, columns=["iso3c", "year", "indicator_code", "value"])
            frames.append(df)

    if not frames:
        return None

    combined = pd.concat(frames, ignore_index=True)
    combined = combined.dropna(subset=["iso3c", "year", "value"])
    return combined


def pivot_wb_wide(
    long_df: pd.DataFrame,
    indicator_mapping: dict[str, str] | None = None,
) -> pd.DataFrame:
    """Pivot WB long-format into wide, using friendly column names.

    indicator_mapping: {friendly_name: indicator_code}
    """
    if indicator_mapping is None:
        indicator_mapping = WB_INDICATORS

    # Filter to only the indicators we need
    codes_needed = set(indicator_mapping.values())
    filtered = long_df[long_df["indicator_code"].isin(codes_needed)].copy()

    # Create reverse mapping: code → friendly name
    code_to_name = {v: k for k, v in indicator_mapping.items()}
    filtered["indicator_name_short"] = filtered["indicator_code"].map(code_to_name)

    # Pivot
    wide = filtered.pivot_table(
        index=["iso3c", "year"],
        columns="indicator_name_short",
        values="value",
        aggfunc="first",
    ).reset_index()

    wide.columns.name = None
    return wide


def read_ihme_dalys(silver_dir: Path | None = None, reference_dir: Path | None = None) -> pd.DataFrame | None:
    """Read IHME GBD DALYs from native Parquet.

    Returns DataFrame with columns: iso3c, year, cause_id, cause_name, dalys
    """
    if silver_dir is None:
        silver_dir = IHME_SILVER
    if reference_dir is None:
        reference_dir = IHME_REFERENCE

    dalys_path = silver_dir / "gbd_results" / "native" / "dalys.parquet"
    if not dalys_path.exists():
        return None

    df = pd.read_parquet(dalys_path)

    # Filter to: both sexes, all ages, number metric
    df = df[
        (df["sex_id"] == 3) &  # Both sexes
        (df["age_id"] == 22) &  # All ages
        (df["metric_id"] == 1)  # Number
    ].copy()

    # Join with location crosswalk for iso3c
    crosswalk_path = reference_dir / "location_to_iso3.csv"
    if crosswalk_path.exists():
        loc_map = pd.read_csv(crosswalk_path)
        df = df.merge(loc_map[["location_id", "iso3c"]], on="location_id", how="left")
    else:
        # Fallback: location_name as-is (less reliable)
        df["iso3c"] = None

    result = df[["iso3c", "year", "cause_id", "cause_name", "val"]].copy()
    result = result.rename(columns={"val": "dalys"})
    result = result.dropna(subset=["iso3c", "dalys"])

    return result


def read_who_health_system(silver_dir: Path | None = None) -> pd.DataFrame | None:
    """Read WHO health system indicators (World Health Statistics + HIDR).

    Returns DataFrame with columns: iso3c, year, indicator_code, value
    """
    if silver_dir is None:
        silver_dir = WHO_SILVER

    frames = []

    # World Health Statistics
    whs_dir = silver_dir / "world_health_statistics"
    if whs_dir.is_dir():
        for pq in whs_dir.glob("*.parquet"):
            df = pd.read_parquet(pq)
            if "dim_geo_code" in df.columns and "value_numeric" in df.columns:
                mapped = pd.DataFrame({
                    "iso3c": df["dim_geo_code"],
                    "year": pd.to_numeric(df.get("dim_time_year", pd.Series(dtype="Int64")), errors="coerce").astype("Int64"),
                    "indicator_code": df.get("ind_code", ""),
                    "value": pd.to_numeric(df["value_numeric"], errors="coerce"),
                })
                frames.append(mapped.dropna(subset=["iso3c", "value"]))

    # HIDR
    hidr_dir = silver_dir / "hidr"
    if hidr_dir.is_dir():
        for pq in hidr_dir.glob("*.parquet"):
            df = pd.read_parquet(pq)
            if "iso3" in df.columns and "estimate" in df.columns:
                mapped = pd.DataFrame({
                    "iso3c": df["iso3"],
                    "year": pd.to_numeric(df.get("date", pd.Series(dtype="Int64")), errors="coerce").astype("Int64"),
                    "indicator_code": df.get("indicator_abbr", ""),
                    "value": pd.to_numeric(df["estimate"], errors="coerce"),
                })
                frames.append(mapped.dropna(subset=["iso3c", "value"]))

    if not frames:
        return None

    return pd.concat(frames, ignore_index=True)
```

```python
# src/shifaa/lakehouse/join.py
"""Build the unified shifaa_matrix from all data sources."""
from __future__ import annotations

import numpy as np
import pandas as pd


def build_shifaa_matrix(
    dalys_df: pd.DataFrame,
    trials_df: pd.DataFrame,
    wb_wide_df: pd.DataFrame,
    who_df: pd.DataFrame | None = None,
) -> pd.DataFrame:
    """Build the shifaa_matrix: one row per country × disease × year.

    Args:
        dalys_df: IHME DALYs (iso3c, year, cause_id, cause_name, dalys)
        trials_df: CT.gov trial density (iso3c, year, gbd_cause_id, trial_density, trial_count_binary)
        wb_wide_df: WB indicators pivoted wide (iso3c, year, gdp_pc, governance, ...)
        who_df: Optional WHO indicators (iso3c, year, indicator_code, value)

    Returns:
        shifaa_matrix DataFrame
    """
    # Start with DALYS as the backbone — every country-cause-year with burden
    matrix = dalys_df.rename(columns={"cause_id": "gbd_cause_id"}).copy()

    # Left-join trial density (fills NaN for missing = no trials)
    matrix = matrix.merge(
        trials_df,
        on=["iso3c", "year", "gbd_cause_id"],
        how="left",
    )
    matrix["trial_density"] = matrix["trial_density"].fillna(0)
    matrix["trial_count_binary"] = matrix["trial_count_binary"].fillna(0).astype(int)

    # Left-join WB indicators
    matrix = matrix.merge(wb_wide_df, on=["iso3c", "year"], how="left")

    # Left-join WHO if available
    if who_df is not None and not who_df.empty:
        # Pivot WHO to wide on indicator_code (take first value per country-year)
        who_wide = who_df.pivot_table(
            index=["iso3c", "year"],
            columns="indicator_code",
            values="value",
            aggfunc="first",
        ).reset_index()
        who_wide.columns.name = None
        matrix = matrix.merge(who_wide, on=["iso3c", "year"], how="left")

    # Compute REI
    population = matrix.get("population", pd.Series(dtype="float64"))
    if population.notna().any():
        dalys_per_100k = matrix["dalys"] / (population / 100_000)
        trial_density_per_100k = matrix["trial_density"] / (population / 100_000)
    else:
        dalys_per_100k = matrix["dalys"]
        trial_density_per_100k = matrix["trial_density"]

    matrix["dalys_per_100k"] = dalys_per_100k

    # REI = log10(trial_density_per_100k / dalys_per_100k)
    # Handle zeros: add small epsilon to avoid log(0)
    eps = 1e-10
    ratio = (trial_density_per_100k + eps) / (dalys_per_100k + eps)
    matrix["rei_score"] = np.log10(ratio)

    # Cap extreme values
    matrix["rei_score"] = matrix["rei_score"].clip(-10, 10)

    return matrix
```

```python
import pandas as pd


def rank_evidence_deserts(matrix: pd.DataFrame, top_n: int = 50, year: int | None = None) -> pd.DataFrame:
    """Rank country-disease pairs by REI (ascending = worst deserts first).

    Args:
        matrix: The shifaa_matrix
        top_n: Number of worst deserts to return
        year: Filter to specific year (default: most recent)

    Returns:
        DataFrame sorted by REI ascending, top_n rows
    """

        df = df[df["year"] == year]
    elif "year" in df.columns:
        df = df[df["year"] == df["year"].max()]

    return (
        df[["iso3c", "year", "gbd_cause_id", "cause_name", "dalys", "trial_density", "rei_score"]]
        .sort_values("rei_score", ascending=True)
        .head(top_n)
        .reset_index(drop=True)
    )


def summarize_rei_by_country(matrix: pd.DataFrame, year: int | None = None) -> pd.DataFrame:
    """Summarize REI per country (mean across diseases).

    if year is not None:
        df = df[df["year"] == year]

        median_rei=("rei_score", "median"),
        n_diseases=("gbd_cause_id", "nunique"),
        total_dalys=("dalys", "sum"),
        total_trials=("trial_density", "sum"),

        mean_rei=("rei_score", "mean"),
        n_countries=("iso3c", "nunique"),
```

```python
# src/shifaa/analysis/regression.py
"""Multilevel Poisson regression for Paper 2 (The Balance)."""
from __future__ import annotations

import numpy as np
import pandas as pd
import statsmodels.api as sm
import statsmodels.formula.api as smf


def fit_poisson_model(matrix: pd.DataFrame) -> dict | None:
    """Fit a Poisson regression: trial_count ~ burden + socioeconomic covariates.

    Uses GEE with exchangeable correlation for country clusters as a
    practical approximation to full multilevel Poisson (statsmodels doesn't
    have a built-in multilevel Poisson, but GEE with cluster gives robust SEs).

    Returns dict with model results or None if fitting fails.
    """
    df = matrix.dropna(subset=[
        "trial_count_binary", "dalys", "gdp_pc", "governance",
        "health_spend_pct", "hci", "physicians", "population",
    ]).copy()

    if len(df) < 20:
        return None

    # Log-transform skewed predictors
    df["log_dalys"] = np.log(df["dalys"].clip(lower=1))
    df["log_gdp_pc"] = np.log(df["gdp_pc"].clip(lower=1))
    df["log_population"] = np.log(df["population"].clip(lower=1))

    try:
        model = smf.gee(
            "trial_count_binary ~ log_dalys + log_gdp_pc + governance + health_spend_pct + hci + physicians",
            groups="iso3c",
            data=df,
            family=sm.families.Poisson(),
            cov_struct=sm.cov_struct.Exchangeable(),
            offset=df["log_population"],
        )
        result = model.fit()
        return {
            "model": result,
            "converged": True,
            "n_obs": len(df),
            "n_countries": df["iso3c"].nunique(),
        }
    except Exception:
        return None


def extract_coefficients(result: dict) -> pd.DataFrame:
    """Extract coefficient table from fitted model.

    Returns DataFrame: variable, coefficient, std_err, z_value, p_value, ci_lower, ci_upper
    """
    model = result["model"]
    summary = model.summary2().tables[1]

    coefs = pd.DataFrame({
        "variable": summary.index,
        "coefficient": summary["Coef."].values,
        "std_err": summary["Std.Err."].values,
        "z_value": summary["z"].values,
        "p_value": summary["P>|z|"].values,
        "ci_lower": summary["[0.025"].values,
        "ci_upper": summary["0.975]"].values,
    }).reset_index(drop=True)

    # Remove intercept for cleaner output
    coefs = coefs[coefs["variable"] != "Intercept"].reset_index(drop=True)

    return coefs
```

```python
# src/shifaa/analysis/equity_trend.py
"""Gini coefficient and Lorenz curve for research equity trends (Paper 5)."""
from __future__ import annotations

import numpy as np
import pandas as pd


def compute_gini(values: np.ndarray) -> float:
    """Compute Gini coefficient for a 1D array of non-negative values."""
    values = np.asarray(values, dtype=float)
    values = values[~np.isnan(values)]
    if len(values) == 0 or values.sum() == 0:
        return 0.0
    sorted_vals = np.sort(values)
    n = len(sorted_vals)
    index = np.arange(1, n + 1)
    return (2 * np.sum(index * sorted_vals) / (n * np.sum(sorted_vals))) - (n + 1) / n


def compute_lorenz(values: np.ndarray) -> tuple[np.ndarray, np.ndarray]:
    """Compute Lorenz curve coordinates.

    Returns (population_fraction, cumulative_share) arrays, both starting at 0 and ending at 1.
    """
    values = np.asarray(values, dtype=float)
    values = values[~np.isnan(values)]
    sorted_vals = np.sort(values)
    cumulative = np.cumsum(sorted_vals)
    total = cumulative[-1] if len(cumulative) > 0 else 1

    pop_frac = np.concatenate([[0], np.arange(1, len(sorted_vals) + 1) / len(sorted_vals)])
    income_frac = np.concatenate([[0], cumulative / total])

    return pop_frac, income_frac


def compute_annual_gini(
    matrix: pd.DataFrame,
    value_col: str = "trial_density",
) -> pd.DataFrame:
    """Compute Gini coefficient of trial density per year.

    Returns DataFrame: year, gini, n_countries
    """
    rows = []
    for year, group in matrix.groupby("year"):
        # Aggregate trial density per country (sum across diseases)
        country_totals = group.groupby("iso3c")[value_col].sum().values
        g = compute_gini(country_totals)
        rows.append({"year": year, "gini": g, "n_countries": len(country_totals)})

    return pd.DataFrame(rows).sort_values("year").reset_index(drop=True)
```

```python
"""REI projection to future years (Paper 3 — The Forecast)."""
from __future__ import annotations

import numpy as np
import pandas as pd


def project_rei(
    current: pd.DataFrame,
    target_year: int = 2036,
    default_dalys_growth: float = 0.02,
    default_trial_growth: float = 0.01,
) -> pd.DataFrame:
    """Project REI forward using burden and trial growth rates.

    If growth rate columns exist in current, uses those.
    Otherwise uses defaults (2% burden growth, 1% trial growth — conservative).
    """
    df = current.copy()
    base_year = df["year"].max() if "year" in df.columns else 2020
    years_ahead = target_year - base_year

    # Get growth rates
    dalys_growth = df.get("dalys_growth_rate", pd.Series(default_dalys_growth, index=df.index))
    trial_growth = df.get("trial_growth_rate", pd.Series(default_trial_growth, index=df.index))

    # Project
    projected_dalys = df["dalys"] * (1 + dalys_growth) ** years_ahead
    projected_trials = df["trial_density"] * (1 + trial_growth) ** years_ahead

    # Recompute REI
    eps = 1e-10
    population = df.get("population", pd.Series(1.0, index=df.index))
    dalys_rate = projected_dalys / (population / 100_000)
    trial_rate = projected_trials / (population / 100_000)
    rei_projected = np.log10((trial_rate + eps) / (dalys_rate + eps))

    result = df[["iso3c", "gbd_cause_id"]].copy()
    result["year"] = target_year
    result["rei_current"] = df["rei_score"]
    result["rei_projected"] = rei_projected.clip(-10, 10)
    result["dalys_projected"] = projected_dalys
    result["rei_change"] = result["rei_projected"] - result["rei_current"]

    return result
```

```python
# src/shifaa/analysis/counterfactual.py
"""Counterfactual scenarios for Paper 4 (The Path / Siraat)."""
from __future__ import annotations

import numpy as np
import pandas as pd


def compute_counterfactual(
    deserts: pd.DataFrame,
    governance_target: float = 0.0,
    gov_coefficient: float = 1.5,
) -> pd.DataFrame:
    """Compute counterfactual REI if governance improved to target level.

    Uses the regression coefficient from Paper 2 to predict how trial
    density would change if governance moved to the target.

    Args:
        deserts: DataFrame with current state (must have governance, trial_density, dalys, population)
        governance_target: Target governance score
        gov_coefficient: Coefficient from Poisson regression (from Paper 2)

    Returns:
        DataFrame with counterfactual columns added
    """
    df = deserts.copy()

    # Change in governance
    gov_change = governance_target - df["governance"]

    # Predicted multiplicative change in trial rate (Poisson model: exp(beta * delta_x))
    trial_multiplier = np.exp(gov_coefficient * gov_change)
    df["trial_density_counterfactual"] = df["trial_density"] * trial_multiplier

    # Recompute REI
    eps = 1e-10
    population = df.get("population", pd.Series(1.0, index=df.index))
    dalys_rate = df["dalys"] / (population / 100_000)
    trial_rate_cf = df["trial_density_counterfactual"] / (population / 100_000)
    df["rei_counterfactual"] = np.log10((trial_rate_cf + eps) / (dalys_rate + eps)).clip(-10, 10)

    df["governance_target"] = governance_target
    df["trial_multiplier"] = trial_multiplier

    return df
```
